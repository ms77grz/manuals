# sh archfi

1 Language
    English
2 Set Keyboard Layout
    us
3 Editor
    vim
    export EDITOR=vim

4 Disk Partitions
    Auto Partitions (dos)
        /dev/sda/ 20G
            /dev/sda1   512M    boot    linux 83
            /dev/sda2   4G      swap    linux 82
            /dev/sda3   15.6G   root    linux 83

5 Select Partitions and Install
    Select boot device:
        /dev/sda1
    Select swap device:
        /dev/sda2
    Select root device:
        /dev/sda3
    Select home device:
        none    -
    Format and Mount
        boot    /dev/sda1   ext4
        swap    /dev/sda2   swap
        root    /dev/sda3   ext4

        mount /dev/sda3 /mnt
        mkdir /mnt/{boot,home}
        mount /dev/sda1 /mnt/boot
        swapon /dev/sda2

6 Install Menu
    Install Arch Linux  pacstrap base kernel
        linux
    Install Firmware pacstrap linux-firmare
    Config Arch Linux
        Set Computer Name   /etc/hostname
            echo "mxsh" > /mnt/etc/hostname
        Set Keyboard Layout /etc/vconsole.conf
            us
                echo "KEYMAP=us" > /mnt/etc/vconsole.conf
        Set Locale /etc/locale.conf, /etc/locale.gen
            en_US
                echo "LANG=en_US.UTF-8" > /mnt/etc/locale.conf
                echo "LC_COLLATE=C" >> /mnt/etc/locale.conf
                sed -i '/en_US.UTF-8/s/^#//g' /mnt/etc/locale.gen
                arch-chroot /mnt /root
                locale-gen
                exit
        Set Time /etc/localtime
            Europe Moscow
                ln -sf /mnt/usr/share/zoneinfo/Europe/Moscow /mnt/etc/localtime
                Use UTC hardware clock
                    arch-root /mnt /root
                    hwclock --systohc --utc
                    exit
        Set Root Password
            arch-chroot /mnt /root
            passwd root
            exit
        Generate fstab
            UUID    genfstab -U
                genfstab -U -p /mnt > /mnt/etc/fstab
        Bootloader
            grub
                install grub    pacstrap grub (efibootmgr, grub-mkconfig)
                    pacstrap /mnt grub
                    arch-chroot /mnt /root
                    mkdir /boot/grub
                    grub-mkconfig -o /boot/grub/grub.cfg
                    exit
                Install bootloader  grub-install
                    /dev/sda    20G
                        arch-chroot /mnt /root
                        grub-install --target=i386-pc --recheck /dev/sda
                        exit
        Enable dhcpcd   systemctl enable dhcpcd
            pacstrap /mnt dhcpcd
            arch-chroot /mnt /root
            systemctl enable dhcpcd
            exit
